1. Consider the following classes:

    class Employee
{
public:
   Employee();
   void setName(const string& name);
   void setSalary(double salary);
   virtual string getName() const;
   virtual double getSalary() const; // Returns base salary
private:
   string m_name;
   double m_salary;
};

class Manager : public Employee
{
   Manager();
   double setBonus(double amount);
   double getSalary() const; // Returns base salary + bonus
private:
   double m_bonus;
};

    Which of the following are a valid body for a Manager::getSalary function that returns the sum of the base salary and the bonus?

    1. return m_salary + bonus;
      The derived class cannot access the m_salary data member of the base class.
    2. return getBaseSalary() + bonus;
      There is no getBaseSalary member function.
    3. return getSalary() + bonus;
      This is a recursive call.
    4. return Employee::getSalary() + bonus;
      The Employee::getSalary member function returns the base salary.
