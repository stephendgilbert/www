1. Examine the following classes:

    class Employee
{
public:
   Employee();
   void setName(const string& newName);
   void setSalary(double newSalary);
   string getName() const;
   virtual double getSalary() const;
private:
   string m_name;
   double m_salary;
};

class Manager : public Employee
{
   Manager();
   double setBonus(double amount);
   double getSalary() const; // Returns base salary + bonus
private:
   double m_bonus;
};

    Which member functions are inherited in Manager?

    1. void setName(string new_name);
      This member function is defined in Employee and not changed in Manager.
    2. double getSalary() const;
      This member function is overridden in the Manager class.
    3. double setBonus(double bonus_amount);
      This member function is not defined in the Employee class.
    4. double getBonus() const;
      This member function is not defined in either class.
