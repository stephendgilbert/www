1. Consider the Shape and Circle classes: 

    class Shape
{
public:
   void draw() const;
   . . .
};

class Circle : public Shape
{
public:
   void draw() const;
   . . .
};

int main()
{
   Shape* p = new Circle;
   p->draw();
}
    1. The code will execute Circle::draw()
    2. The code will not compile since Shape::draw() is not virtual
    3. The code will execute Shape::draw()
    4. The code will not compile because Shape::draw() is a pure-virtual function.
