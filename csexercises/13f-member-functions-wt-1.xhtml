    struct CashRegister
{
// member functions
   void clear();
   void addItem(double price);
   int getCount() const;
   double getTotal() const;
// data members
   int itemCount;
   double totalPrice;
};
void CashRegister::clear()
{
   itemCount = 0;
   totalPrice = 0;
}
void CashRegister::addItem(double price)
{
   totalPrice = totalPrice + price;
   itemCount++;
}
double CashRegister::getCount() const
{
   return itemCount;
}
double CashRegister::getTotal() const
{
   return totalPrice;
}
