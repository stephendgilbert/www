1. Consider the Person and Student classes: 

    class Person
{
public:
   virtual string toString() const { return "Person"; }
   . . .
};

class Student: public Person
{
public:
   string toString() const { return "Student"; }
   . . .
};

int main()
{
   Person* p = new Student;
   cout << p->toString() << endl;
}

    What happens?

    1. The code will print "Student", since toString uses late binding and is overridden in Student.
    2. The code will not compile since p is a Person* so it must point to a Person object
    3. The code prints "Person" since toString uses early binding to bind to the Person *
    4. The code prints both "Person" and "Student", since p is a Person* which points to a Student object
