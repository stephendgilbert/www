class ChoiceQuestion : public Question
{
public:
   void addChoice(const string& choice, bool correct);
   void display() const;

   vector<string> choices;

void ChoiceQuestion::addChoice(const string& choice, bool correct)
{
   choices.push_back(choice);
   if (correct)
   {
      int index = choices.size();
      string choiceStr = to_string(index);
      set_answer(choiceStr);
   }
}
