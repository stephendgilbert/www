1. Consider the following classes:

    class Employee
{
public:
   Employee();
   Employee(const string& name, double salary);
   void setName(const string& name, double salary);
   void setSalary(double salary);
   virtual string name() const;
   virtual double salary() const; // Returns base salary
private:
   string m_name;
   double m_salary;
};

class Manager : public Employee
{
   Manager(const string& name, double salary, double bonus);
   double salary() const; // Returns base salary + bonus
private:
   double m_bonus;
};

    Which of the following are a correct implementation of the Manager constructor? 

    1. Manager::Manager(const string& name, double salary, double bonus)
{
   m_name = name; m_salary = salary; m_bonus = bonus;
}
      The derived class cannot access the m_salary and m_name data members of the base class.
    2. Manager::Manager(const string& name, double salary, double bonus)
{
   setName(name); setSalary(salary); m_bonus = bonus;
}
      This calls the default Employee constructor first.
    3. Manager::Manager(const string& name, double salary, double bonus)
  : Employee(name, salary, bonus) 
{
}
      There is no Employee constructor taking three arguments.
    4. Manager::Manager(const string& name, double salary, double bonus)
  : Employee(name, salary) 
{ 
   m_bonus = bonus; 
}
      It is preferred to call the Employee constructor using the initializer list.
    5. Manager::Manager(const string& name, double salary, double bonus)
  : Employee(name, salary), m_bonus(bonus) 
{ 
}
      You can also initialize the m_bonus in the initializer list.
