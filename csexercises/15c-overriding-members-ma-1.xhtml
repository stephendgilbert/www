1. Given these classes:

    class Question
{
public:
   virtual void display() const;
   void setAnswer(const string& answer);
   . . .
private:
   string m_text;
   string m_answer;
};
class ChoiceQuestion : public Question
{
public:
   void display() const;
   . . .
};

    Which of these member function implementations are correct?

    1. void ChoiceQuestion::display() const
{
   cout << m_text << endl;
   for (int i = 0; i < choices.size(); i++)
   {
      cout << i + 1 << ": " << choices[i] << endl;
   }
}
      The ChoiceQuestion class has no access to the private data member m_text of the Question class.
    2. void ChoiceQuestion::display() const
{
   this->display();
   for (int i = 0; i < choices.size(); i++)
   {
      cout << i + 1 << ": " << choices[i] << endl;
   }
}
      The type of the this pointer is ChoiceQuestion*. Therefore, the display function of ChoiceQuestion is selected, and the function calls itself.
    3. void ChoiceQuestion::display() const
{
   Question::display();
   for (int i = 0; i < choices.size(); i++)
   {
      cout << i + 1 << ": " << choices[i] << endl;
   }
}
      Now the correct display function is called.
    4. void ChoiceQuestion::addChoice(const string& choice, bool correct)
{
   choices.push_back(choice);
   if (correct)
   {
      string numStr = to_string(choices.size());
      Question::setAnswer(numStr);
   }
}
      This implementation is correct, but it was not necessary to add Question:: before setAnswer because setAnswer is not overridden in ChoiceQuestion.
