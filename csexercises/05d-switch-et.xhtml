    Expression Output Explanation 
    int n{1};
switch (n)
{
  case 1: cout << 1;
  case 2: cout << 2;
}
    		12 There is no break statement 
    int n{1};
switch (n)
{
  case 1: cout << 1; break;
  case 2: cout << 2; break;
}
    		1 The break statement means only the first line prints. 
    double n{1.0};
switch (n)
{
  case 1.0: cout << 1.0; break;
  case 2.0: cout << 2.0; break;
}
    		error You must use an integral type for a switch selector. 
    char c = 'X';
switch (c)
{
  case 'A': cout << 'A'; break;
  case 'B': cout << 'B'; break;
  default: cout << c; break;
}
    		X char is an integral type so it can be used for a switch selector. The selector does not match either case so the default is selected.
