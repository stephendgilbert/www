13: import java.awt.*;

15: public class Dot extends Component
16: {
17:     private int radius = 10;
18:     
19:     /**
20:      * Change the radius of the dot
21:      * @param newRadius - new radius
22:      */
23:     public void setRadius(int newRadius)
24:     {
25:         radius = newRadius;
26:     }
27:     
28:     /**
29:      * Paints the component when its drawn
30:      */
31:     public void paint(Graphics g)
32:     {
33:         g.fillOval(0, 0, getWidth(), getHeight());
34:     }
35: }
