01: /**
02:  *  Illustrates nested loops
03:  *  Example for CS 170
04:  *
05:  *  @author Stephen Gilbert
06:  *  @version 2.0, Fall 2005
07:  */
08: import java.awt.*;
09: import java.awt.event.*;
10: import java.applet.*;
11: 
12: public class Times extends Applet
13: {
14:     private TextArea output = new TextArea(10, 40);
15:     private Button   go     = new Button("Go");
16: 
17:     public void init()
18:     {
19:         setLayout(new BorderLayout());
20:         add(go,     BorderLayout.NORTH);
21:         add(output, BorderLayout.CENTER);
22:         output.setFont(new Font("Monospaced", Font.PLAIN, 10));
23:         go.addActionListener(new TablePrinter());
24:     }
25: 
26:     /**
27:      *  Prints a times table when pressed
28:      */
29:     private class TablePrinter implements ActionListener
30:     {
31:         public void actionPerformed(ActionEvent ae)
32:         {
33:             for (int outer = 1; outer <= 10; outer++)
34:             {
35:                 for (int inner = 1; inner <= 10; inner++)
36:                 {
37:                     int product = inner * outer;
38: 
39:                     if (product < 10)
40:                       output.append("   " + product);
41:                     else if (product < 100)
42:                       output.append("  " + product);
43:                     else
44:                       output.append(" " + product);
45:                 }
46:                 output.append("\n");
47:             }
48:         }
49:     }
50: }
