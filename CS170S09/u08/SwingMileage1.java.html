001: /**
002:  *  SwingMileage.java
003:  *
004:  *  Uses JTextField and JFormattedTextField for
005:  *  input of numbers.
006:  *
007:  *  @author Steve Gilbert
008:  *  @version 2.0 September 2005
009:  */
010: import java.awt.*;
011: import java.awt.event.*;
012: import javax.swing.*;
013: import java.text.*;
014: 
015: public class SwingMileage extends JPanel
016: {
017:     // Formatters for both input and output
018:     private NumberFormat nfmt = NumberFormat.getInstance();
019:     private NumberFormat cfmt = NumberFormat.getCurrencyInstance();
020: 
021:     private JTextField startMilesTF = new JTextField();
022:     private JTextField endMilesTF   = new JTextField();
023:     private JTextField fillupCostTF = new JTextField();
024:     private JTextField gallonsTF    = new JTextField();
025: 
026:     private JLabel milesPerGallonLbl = new JLabel("", JLabel.RIGHT);
027:     private JLabel costPerMileLbl    = new JLabel("", JLabel.RIGHT);
028: 
029:     public SwingMileage()
030:     {
031:         Panel p = new Panel();
032:         p.setLayout(new GridLayout(0, 2, 5, 5));
033: 
034:         // Align the text fields to the right
035:         startMilesTF.setHorizontalAlignment(JTextField.TRAILING);
036:         endMilesTF.setHorizontalAlignment(JTextField.TRAILING);
037:         fillupCostTF.setHorizontalAlignment(JTextField.TRAILING);
038:         gallonsTF.setHorizontalAlignment(JTextField.TRAILING);
039: 
040:         // Add input items to the panel
041:         p.add(new JLabel("Beginning mileage :", JLabel.RIGHT));
042:         p.add(startMilesTF);
043:         p.add(new JLabel("Current mileage :", JLabel.RIGHT));
044:         p.add(endMilesTF);
045:         p.add(new JLabel("Cost of fill-up :", JLabel.RIGHT));
046:         p.add(fillupCostTF);
047:         p.add(new JLabel("Gallons purchased :", JLabel.RIGHT));
048:         p.add(gallonsTF);
049: 
050:         // Hook up the last textfield
051:         gallonsTF.addActionListener(new MileageCalculator());
052: 
053:         // Output area
054:         p.add(new JLabel("Miles per gallon :", JLabel.RIGHT));
055:         p.add(milesPerGallonLbl);
056:         p.add(new JLabel("Cost per mile :", JLabel.RIGHT));
057:         p.add(costPerMileLbl);
058: 
059:         milesPerGallonLbl.setFont(new Font("Serif", Font.BOLD, 14));
060:         costPerMileLbl.setFont(new Font("Serif", Font.BOLD, 14));
061:         milesPerGallonLbl.setForeground(Color.red);
062:         costPerMileLbl.setForeground(Color.red);
063: 
064:         // Add the panel to the app
065:         add(p);
066:     }
067:     /**
068:      *  MileageCalculator class performs the calculations
069:      */
070:     private class MileageCalculator implements ActionListener
071:     {
072:         public void actionPerformed(ActionEvent ae)
073:         {
074:             // Extract values from each textfield
075:             String startMilesStr = startMilesTF.getText();
076:             String endMilesStr   = endMilesTF.getText();
077:             String fillupCostStr = fillupCostTF.getText();
078:             String gallonsStr    = gallonsTF.getText();
079: 
080:             // Convert to numbers
081:             double startMiles = Double.parseDouble(startMilesStr);
082:             double endMiles   = Double.parseDouble(endMilesStr);
083:             double fillupCost = Double.parseDouble(fillupCostStr);
084:             double gallons    = Double.parseDouble(gallonsStr);
085: 
086:             // Do the calculation
087:             double miles = (endMiles - startMiles);
088:             double mpg = miles / gallons;
089:             double cpm = fillupCost / miles;
090: 
091:             // Display output, using correct formatting
092:             milesPerGallonLbl.setText(nfmt.format(mpg));
093:             costPerMileLbl.setText(cfmt.format(cpm));
094:         }
095:     }
096: 
097:     /**
098:      * Creates the JFrame and starts the app running
099:      *
100:     */
101:     public static void main(String[] args)
102:     {
103:         // Set the UI to the system look and feel
104:         try { UIManager.setLookAndFeel(
105:             UIManager.getSystemLookAndFeelClassName() );
106:         }
107:         catch ( Exception e ) {}
108: 
109:         JFrame frame = new JFrame ("Swing Mileage");
110:         frame.setDefaultCloseOperation (JFrame.EXIT_ON_CLOSE);
111:         frame.getContentPane().add(new SwingMileage());
112:         frame.pack();
113:         frame.setVisible(true);
114:     }
115: }
