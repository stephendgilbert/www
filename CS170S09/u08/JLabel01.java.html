01: /**
02:  * Displays a few JLabels using different constructors
03:  *
04:  * @author Stephen Gilbert
05:  * @version 1.0, September 2005
06:  */
07: import java.awt.*;
08: import java.awt.event.*;
09: import javax.swing.*;
10: 
11: public class JLabel01 extends JPanel
12: {
13:     /**
14:      * Constructor for objects of class JLabel01
15:      */
16:     public JLabel01()
17:     {
18:         setBackground(new Color(196, 255, 196));
19:         Dimension labelSize = new Dimension(100, 100);
20: 
21:         // 1. A plain label with centered text
22:         JLabel lab1 = new JLabel("A Plain Label", JLabel.CENTER);
23:         lab1.setPreferredSize(labelSize);
24:         lab1.setOpaque(true);
25:         lab1.setBackground(Color.yellow);
26:         add(lab1);
27: 
28:         // 2. An image-only icon
29:         JLabel lab2 = new JLabel(new ImageIcon("Car1.GIF"));
30:         lab2.setPreferredSize(labelSize);
31:         lab2.setOpaque(true);
32:         lab2.setBackground(Color.cyan);
33:         add(lab2);
34: 
35:         // Set the position of the icon 1
36:         lab2.setHorizontalAlignment(JLabel.LEFT);
37:         lab2.setVerticalAlignment(JLabel.BOTTOM);
38: 
39:         // 3. An image and text icon
40:         JLabel lab3 = new JLabel("Car2.GIF",
41:                     new ImageIcon("Car2.GIF"), JLabel.RIGHT);
42:         lab3.setPreferredSize(labelSize);
43:         lab3.setOpaque(true);
44:         lab3.setBackground(Color.pink);
45:         add(lab3);
46: 
47:         // Set the position of the icon 3 and the text
48:         lab3.setVerticalAlignment(JLabel.TOP);
49:         lab3.setVerticalTextPosition(JLabel.BOTTOM);
50:         lab3.setHorizontalAlignment(JLabel.CENTER);
51:         lab3.setHorizontalTextPosition(JLabel.CENTER);
52:         lab3.setIconTextGap(10);
53:     }
54: 
55:     /**
56:      * The main() method creates the JFrame and the application panel
57:      */
58:     public static void main(String[] args)
59:     {
60:         JFrame frame = new JFrame("JLabel01");
61:         frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
62:         frame.getContentPane().add(new JLabel01());
63:         frame.setSize(350, 150);
64:         frame.setVisible(true);
65:     }
66: }
