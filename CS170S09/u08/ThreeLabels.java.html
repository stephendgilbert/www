01: /**
02:  *  ThreeLabels.java
03:  *
04:  *  An applet to change the size of a Font and
05:  *  explore its effect on Label size. An example for CS 170.
06:  *
07:  *  @author Stephen Gilbert
08:  *  @version 2.0, September 2005
09:  */
10: import java.awt.*;
11: import java.applet.*;
12: 
13: public class ThreeLabels extends Applet
14: {
15:     private Label smallLbl   = new Label("Small Font", Label.CENTER);
16:     private Label defaultLbl = new Label("Default Font", Label.LEFT);
17:     private Label largeLbl   = new Label("Large Font", Label.RIGHT);
18: 
19:     public void init()
20:     {
21:         this.setLayout(null);               // turn off positioning
22: 
23:         // Calculate sizes
24:         int width = this.getSize().width;   // applet width
25:         int height = this.getSize().height; // applet height
26:         int h1 = height / 10;               // smallest label
27:         int h2 = h1 * 3;                    // default label
28:         int h3 = height - h1 - h2;          // big label
29: 
30:         // Set the bounds
31:         smallLbl.setBounds(0, 0, width, h1);
32:         defaultLbl.setBounds(0, h1, width, h2);
33:         largeLbl.setBounds(0, h1 + h2, width, h3);
34: 
35:         this.add(smallLbl);
36:         this.add(defaultLbl);
37:         this.add(largeLbl);
38: 
39:         // Change the applet's background color
40:         this.setBackground(Color.red);
41: 
42:         // Change the Font used by two Labels
43:         smallLbl.setFont(new Font("SansSerif", Font.PLAIN, 5));
44:         largeLbl.setFont(new Font("Serif", Font.BOLD, 72));
45:     }
46: }
