01: /**
02:  *  TestColor.java
03:  *
04:  *  An applet to test the use of setForeground()
05:  *  and setBackground() on several AWT components.
06:  *
07:  *  @author Stephen Gilbert
08:  *  @version 2.0, September 2005
09:  */
10: 
11: import java.awt.*;
12: import java.applet.*;
13: 
14: public class TestColor extends Applet
15: {
16:     private Label     lbl1 = new Label("Red FG");
17:     private Label     lbl2 = new Label("Red BG");
18:     private Button    btn1 = new Button("Green FG");
19:     private Button    btn2 = new Button("Green BG");
20:     private List      lst1 = new List();
21:     private List      lst2 = new List();
22:     private Scrollbar sb1  = new Scrollbar(Scrollbar.HORIZONTAL);
23:     private Scrollbar sb2  = new Scrollbar(Scrollbar.HORIZONTAL);
24: 
25:     public void init()
26:     {
27:         lst1.add("Yellow FG");
28:         lst2.add("Yellow BG");
29: 
30:         lbl1.setForeground(Color.red);
31:         lbl2.setBackground(Color.red);
32:         btn1.setForeground(Color.green);
33:         btn2.setBackground(Color.green);
34:         lst1.setForeground(Color.yellow);
35:         lst2.setBackground(Color.yellow);
36:         sb1.setForeground(Color.pink);
37:         sb2.setBackground(Color.pink);
38: 
39:         setLayout(new GridLayout(5,2,5,5));
40:         add(lbl1);
41:         add(lbl2);
42:         add(btn1);
43:         add(btn2);
44:         add(lst1);
45:         add(lst2);
46:         add(sb1);
47:         add(sb2);
48:     }
49: }
