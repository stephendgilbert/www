  7 /**
  8    Removes an element from an ordered vector.
  9    @param values a vector
 10    @param pos the position of the element to erase
 11 */
 12 void erase(vector<string>& values, int pos)
 13 {
 14    for (int i = pos; i < values.size() - 1; i++)
 15       values[i] = values[i + 1];
 16    values.pop_back();
 17 }
 18 
 19 /**
 20    Prints all elements in a vector.
 21    @param values the vector to print
 22 */
 23 void print(vector<string> values)
 24 {
 25    for (int i = 0; i < values.size(); i++)
 26       cout << "[" << i << "] " << values[i] << "\n";
 27 }
