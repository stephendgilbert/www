  5 /**
  6    Reads data into an array, using 0 as a sentinel value.
  7    @param a the array to fill
  8    @param a_capacity the maximum size of a
  9    @param a_size filled with the size of a after reading the input
 10  */
 11 void read_data(double a[], int a_capacity, int& a_size)
 12 {
 13    a_size = 0;
 14 
 15    bool more = true;
 16    while (more)
 17    {
 18       double x;
 19       cin >> x;
 20       if (x == 0)
 21          more = false;
 22       else if (a_size == a_capacity)
 23       {
 24          cout << "Sorry--extra data ignored\n";
 25          more = false;
 26       }
 27       else
 28       {
 29          a[a_size] = x;
 30          a_size++;
 31       }
 32    }
 33 }
 34 
 35 /**
 36  Computes the maximum value in an array.
 37  @param a the array
 38  @param a_size the number of values in a
 39  */
 40 double maximum(const double a[], int a_size)
 41 {
 42    if (a_size == 0)
 43       return 0;
 44    double highest = a[0];
 45    for (int i = 1; i < a_size; i++)
 46       if (a[i] > highest)
 47          highest = a[i];
 48    return highest;
 49 }
