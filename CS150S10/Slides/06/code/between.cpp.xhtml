  6 /**
  7    Returns all values within a range.
  8    @param values a vector of floating-point numbers
  9    @param low the low end of the range
 10    @param high the high end of the range
 11    @return a vector of the given values that fall within the range
 12 */
 13 vector<double> between(vector<double> values,
 14     double low, double high)
 15 {
 16    vector<double> result;
 17    for (int i = 0; i < values.size(); i++)
 18       if (low <= values[i] && values[i] <= high)
 19          result.push_back(values[i]);
 20    return result;
 21 }
