  2 #include <string>

  4 
  5 using namespace std;
  6 
  7 int main()
  8 {
  9    vector<string> names;

 11    vector<int> scores;
 12 
 13    double best_price = 1;
 14    int best_score = 0;
 15    int best_index = -1;
 16 
 17    bool more = true;
 18    while (more)
 19    {
 20       string next_name;

 37          best_index = names.size() - 1;

 40       }
 41 
 42       cout << "More data? (y/n) ";
 43       string answer;
 44       getline(cin, answer);

 46    }
 47 
 48    for (int i = 0; i < names.size(); i++)
 49    {
 50       cout << names[i]
 51          << " Price: " << prices[i]
 52          << " Score: " << scores[i];
 53       if (i == best_index) cout << " <= best value";
 54       cout << "\n";
 55    }
 56

 58 }
