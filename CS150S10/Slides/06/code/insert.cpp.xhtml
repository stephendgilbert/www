  5 using namespace std;

  7 /**
  8    Inserts an element into a vector.
  9    @param values a vector
 10    @param pos the position before which to insert the element
 11    @param s the element to insert
 12 */
 13 void insert(vector<string>& values, int pos, string s)
 14 {
 15    int last = values.size() - 1;
 16    values.push_back(values[last]);
 17    for (int i = last; i > pos; i--)
 18       values[i] = values[i - 1];
 19    values[pos] = s;
 20 }
 21 
 22 /**
 23    Prints all elements in a vector.
 24    @param values the vector to print
 25 */
 26 void print(vector<string> values)
 27 {
 28    for (int i = 0; i < values.size(); i++)
 29       cout << "[" << i << "] " << values[i] << "\n";
 30 }
