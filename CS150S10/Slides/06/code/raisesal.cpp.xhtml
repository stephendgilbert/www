  6 /**
  7    Raise all values in a vector by a given percentage.
  8    @param values vector of values
  9    @param rate percentage rate to raise values by
 10 */
 11 void raise_by_percent(vector<double>& values, double rate)
 12 {
 13    for (int i = 0; i < values.size(); i++)
 14       values[i] = values[i] * (1 + rate / 100);
 15 }
 16
