  6 /**
  7    Computes the average of a vector of floating-point values.
  8    @param values a vector of floating-point values
  9    @return the average of the values
 10 */
 11 double average(vector<double> values)
 12 {
 13    if (values.size() == 0) return 0;
 14    double sum = 0;
 15    for (int i = 0; i < values.size(); i++)
 16       sum = sum + values[i];
 17    return sum / values.size();
 18 }
 19 
 20 int main()
 21 {
 22    vector<double> salaries(5);
 23    salaries[0] = 35000.0;
 24    salaries[1] = 63000.0;
 25    salaries[2] = 48000.0;
 26    salaries[3] = 78000.0;
 27    salaries[4] = 51500.0;
 28 
 29    double avgsal = average(salaries);
