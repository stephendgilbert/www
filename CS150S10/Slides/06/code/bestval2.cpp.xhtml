 12 class Product
 13 {
 14 public:
 15    /**
 16       Constructs a product with score 0 and price 1.
 17    */
 18    Product();
 19 
 20    /**
 21       Reads in this product object.
 22    */
 23    void read();
 24 
 25    /**
 26       Compares two product objects.
 27       @param b the object to compare with this object
 28       @return true if this object is better than b
 29    */
 30    bool is_better_than(Product b) const;
 31 
 32    /**
 33       Print this product object
 34    */
 35    void print() const;
 36 private:
 37    string name;
 38    double price;
 39    int score;
 40 };
 41 
 42 Product::Product()
 43 {
 44    price = 1;
 45    score = 0;
 46 }
 47 
 48 void Product::read()
 49 {
 50    cout << "Please enter the model name: ";
 51    getline(cin, name);
 52    cout << "Please enter the price: ";
 53    cin >> price;
 54    cout << "Please enter the score: ";
 55    cin >> score;
 56    string remainder; // Read remainder of line
 57    getline(cin, remainder);
 58 }
 59 
 60 bool Product::is_better_than(Product b) const
 61 {
 62    if (price == 0) return true;
 63    if (b.price == 0) return false;
 64    return score / price > b.score / b.price;
 65 }
 66 
 67 void Product::print() const
 68 {
 69    cout << name
 70       << " Price: " << price
 71       << " Score: " << score;
 72 }

 74 int main()
 75 {
 76    vector<Product> products;
 77 
 78    Product best_product;
 79    int best_index = -1;
 80 
 81    bool more = true;
 82    while (more)
 83    {
 84       Product next_product;
 85       next_product.read();
 86       products.push_back(next_product);
 87 
 88       if (next_product.is_better_than(best_product))
 89       {
 90          best_index = products.size() - 1;
 91          best_product = next_product;
 92       }
 93 
 94       cout << "More data? (y/n) ";
 95       string answer;
 96       getline(cin, answer);
 97       if (answer != "y") more = false;
 98    }
 99 
100    for (int i = 0; i < products.size(); i++)
101    {
102       products[i].print();
103       if (i == best_index) cout << " <= best value ";
104       cout << "\n";
105    }
106 
107    return 0;
108 }
