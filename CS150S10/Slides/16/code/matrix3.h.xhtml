 12 /**
 13    This class describes a row in a matrix.
 14 */
 15 template <typename T>
 16 class MatrixRow
 17 {
 18 public:
 19    /**
 20       Remembers a row for a given matrix.
 21       @param m a pointer to the matrix
 22       @param s the size of the row
 23    */
 24    MatrixRow(Matrix<T>* m, int s);
 25 
 26    /**
 27       Accesses a row element.
 28       @param j the column index
 29       @return a reference to the element with the given index
 30    */
 31    T& operator[](int j);
 32 
 33 private:
 34    Matrix<T>* mat;
 35    int i;
 36 };

 64 /**
 65    This class describes a matrix with arbitrary rows and columns.
 66 */
 67 template <typename T>
 68 class Matrix
 69 {
 70 public:
 71    /**
 72       Constructs a matrix filled with zero elements.
 73    */
 74    Matrix(int r, int c);
 75 
 76    /**
 77       The big three-copy constructor, assignment operator, and destructor.
 78    */
 79    Matrix(const Matrix<T>& other);
 80    Matrix& operator=(const Matrix<T>& other);
 81    ~Matrix();
 82 
 83    /**
 84       Gets the number of rows of this matrix.
 85       @return the number of rows
 86    */
 87    int get_rows() const;
 88 
 89    /**
 90       Gets the number of columns of this matrix.
 91       @return the number of columns
 92    */
 93    int get_columns() const;
 94 
 95    /**
 96       Accesses a matrix element.
 97       @param i the row index
 98       @param j the column index
 99       @return a reference to the element with the given indexes
100    */
101    T& operator()(int i, int j);
102 
103    /**
104       Accesses a matrix element.
105       @param i the row index
106       @param j the column index
107       @return the element with the given indexes
108    */
109    T operator()(int i, int j) const;
110 
111    /**
112       Accesses a matrix row.
113       @param i the row index
114       @return the row with the given index
115    */
116    MatrixRow<T> operator[](int i);
117 
118    /**
119       Accesses a matrix row.
120       @param i the row index
121       @return the row with the given index
122    */
123    ConstMatrixRow<T> operator[](int i) const;
124 
125    /**
126       Computes the matrix sum.
127       @param right another matrix
128       @return the updated matrix
129    */
130    Matrix<T>& operator+=(const Matrix<T>& right);
131 
132 private:
133    /**
134       Copies another matrix into this matrix.
135       @param other the other matrix
136    */
137    void copy(const Matrix<T>& other);
138 
139    /**
140       Frees the memory for this matrix.
141    */
142    void free();
143 
144    int rows;
145    int columns;
146    T* elements;
147 };

278 template <typename T>

282    {
283       free();
284       copy(other);
285    }

289 template <typename T>
290 void Matrix<T>::copy(const Matrix<T>& other)
291 {
292    rows = other.rows;
293    columns = other.columns;
294    elements = new T[rows * columns];
295    for (int i = 0; i < rows * columns; i++)
296       elements[i] = other.elements[i];
297 }

299 template <typename T>
300 Matrix<T>& Matrix<T>::operator+=(const Matrix<T>& right)
301 {
302    assert(rows == right.rows && columns == right.columns);
303    for (int i = 0; i < rows; i++)
304       for (int j = 0; j < columns; j++)
305          (*this)(i, j) += right(i, j);
306    return *this;
307 }

309 template <typename T>

311 {
312    Matrix<T> result = left;
313    result += right;
314    return result;
315 }

317 template <typename T>
318 Matrix<T> operator*(const Matrix<T>& left, const Matrix<T>& right)
319 {
320    assert(left.get_columns() == right.get_rows());
321    Matrix<T> result(left.get_rows(), right.get_columns());
322    for (int i = 0; i < left.get_rows(); i++)
323       for (int j = 0; j < right.get_columns(); j++)
324          for (int k = 0; k < left.get_columns(); k++)
325             result(i, j) += left(i, k) * right(k, j);
326    return result;
327 }
328 
329 template <typename T>
330 Matrix<T> operator*(const Matrix<T>& left, T right)
331 {
332    Matrix<T> result(left);
333    for (int i = 0; i < result.get_rows(); i++)
334       for (int j = 0; j < result.get_columns(); j++)
335          result(i, j) *= right;
336    return result;
337 }

339 template <typename T>
340 ostream& operator<<(ostream& left, const Matrix<T>& right)
341 {
342    const int WIDTH = 10;
343    for (int i = 0; i < right.get_rows(); i++)
344    {
345       for (int j = 0; j < right.get_columns(); j++)
346          left << setw(WIDTH) << right(i, j);
347       left << "\n";
348    }
349    return left;
