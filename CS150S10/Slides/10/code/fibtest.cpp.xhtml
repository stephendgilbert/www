  3 using namespace std;
  4 
  5 /**
  6    Computes a Fibonacci number.
  7    @param n an integer
  8    @return the nth Fibonacci number
  9 */
 10 int fib(int n)
 11 {
 12    if (n <= 2) return 1;
 13    else return fib(n - 1) + fib(n - 2);
 14 }
 15 
 16 int main()
 17 {
 18    cout << "Enter n: ";
 19    int n;
 20    cin >> n;
 21    int f = fib(n);
 22    cout << "fib(" << n << ") = " << f << "\n";
 23    return 0;
 24 }
