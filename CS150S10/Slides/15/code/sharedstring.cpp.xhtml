  5 class SharedString

  7 public:

 14    class StringReference;

 19 class SharedString::StringReference
 20 {
 21 public:
 22    int count;
 23    char* buffer;
 24    StringReference(const char* right);
 25    ~StringReference();
 26 };

 28 SharedString::SharedString() : p(NULL)

 30    cout << "Entering SharedString()\n";

 32 
 33 SharedString::SharedString(const char* right) : p(NULL)

 39 SharedString::SharedString(const SharedString& right) : p(NULL)

 47    cout << "Entering ~SharedString() ";

 51 SharedString& SharedString::operator=(const SharedString& right)
 52 {
 53    cout << "Entering operator=(const SharedString&) ";
 54    reassign(right.p);
 55    return *this;

 58 void SharedString::reassign(SharedString::StringReference* np)
 59 {
 60    if (np == NULL) // Print old values
 61       cout << "value: " << p->buffer << " count: " << p->count << "\n";
 62    if (np != NULL) // Increment count on the new value
 63       np->count += 1;
 64    if (p != NULL)  // Decrement reference counts on old value
 65    {
 66       p->count -= 1;
 67       if (p->count == 0)
 68          delete p;
 69    }
 70    p = np;  // Change binding
 71    if (p != NULL) // Print new values
 72       cout << "value: " << p->buffer << " count: " << p->count << "\n";
 73 }
 74 
 75 SharedString::StringReference::StringReference(const char* right)
 76 {
 77    count = 0;
 78    int n = 0;
 79    while (right[n] != '\0')
 80       n++;
 81    buffer = new char[1 + n];
 82    for (int i = 0; i < n; i++)
 83       buffer[i] = right[i];
 84    buffer[n] = '\0';
 85 }
 86 
 87 SharedString::StringReference::~StringReference()
 88 {
 89    delete[] buffer;
 90 }
