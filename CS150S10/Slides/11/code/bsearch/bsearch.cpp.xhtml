  5 /**
  6    Finds an element in a sorted vector.
  7    @param v the sorted vector with the elements to search
  8    @param from the start of the range to search
  9    @param to the end of the range to search
 10    @param value the value to search for
 11    @return the index of the first match, or -1 if not found
 12 */
 13 int binary_search(vector<int> v, int from, int to, int value)
 14 {
 15    if (from > to)
 16       return -1;
 17    int mid = (from + to) / 2;
 18    if (v[mid] == value)
 19       return mid;
 20    else if (v[mid] < value)
 21       return binary_search(v, mid + 1, to, value);
 22    else
 23       return binary_search(v, from, mid - 1, value);
 24 }
