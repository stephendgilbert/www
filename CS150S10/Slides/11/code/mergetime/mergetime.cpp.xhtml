  6 /**
  7    Merges two adjacent ranges in a vector
  8    @param a the vector with the elements to merge
  9    @param from the start of the first range
 10    @param mid the end of the first range
 11    @param to the end of the second range
 12 */
 13 void merge(vector<int>& a, int from, int mid, int to)
 14 {
 15    int n = to - from + 1; // Size of the range to be merged
 16    // Merge both halves into a temporary vector b
 17    vector<int> b(n);
 18 
 19    int i1 = from;
 20       // Next element to consider in the first half
 21    int i2 = mid + 1;
 22       // Next element to consider in the second half
 23    int j = 0; // Next open position in b
 24 
 25    // As long as neither i1 nor i2 is past the end, move the smaller
 26    // element into b
 27    while (i1 <= mid && i2 <= to)
 28    {
 29       if (a[i1] < a[i2])
 30       {
 31          b[j] = a[i1];
 32          i1++;
 33       }
 34       else
 35       {
 36          b[j] = a[i2];
 37          i2++;
 38       }
 39       j++;
 40    }
 41 
 42    // Note that only one of the two while loops below is executed
 43 
 44    // Copy any remaining entries of the first half
 45    while (i1 <= mid)
 46    {
 47       b[j] = a[i1];
 48       i1++;
 49       j++;
 50    }
 51    // Copy any remaining entries of the second half
 52    while (i2 <= to)
 53    {
 54       b[j] = a[i2];
 55       i2++;
 56       j++;
 57    }
 58 
 59    // Copy back from the temporary vector
 60    for (j = 0; j < n; j++)
 61       a[from + j] = b[j];
 62 }
 63 
 64 /**
 65    Sorts the elements in a range of a vector.
 66    @param a the vector with the elements to sort
 67    @param from start of the range to sort
 68    @param to end of the range to sort
 69 */
 70 void merge_sort(vector<int>& a, int from, int to)
 71 {
 72    if (from == to) return;
 73    int mid = (from + to) / 2;
 74    // Sort the first and the second half
 75    merge_sort(a, from, mid);
 76    merge_sort(a, mid + 1, to);
 77    merge(a, from, mid, to);
 78 }

 80 int main()
