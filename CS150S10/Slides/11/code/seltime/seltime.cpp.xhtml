 10 /**
 11    Swaps two integers.
 12    @param x the first integer to swap
 13    @param y the second integer to swap
 14 */
 15 void swap(int& x, int& y)
 16 {
 17    int temp = x;
 18    x = y;
 19    y = temp;
 20 }
 21 
 22 /**
 23     Gets the position of the smallest element in a vector range.
 24     @param a the vector
 25     @param from the beginning of the range
 26     @param to the end of the range
 27     @return the position of the smallest element in
 28     the range a[from]...a[to]
 29 */
 30 int min_position(vector<int>& a, int from, int to)
 31 {
 32    int min_pos = from;
 33    int i;
 34    for (i = from + 1; i <= to; i++)
 35       if (a[i] < a[min_pos]) min_pos = i;
 36    return min_pos;
 37 }

 39 /**
 40    Sorts a vector using the selection sort algorithm
 41    @param a the vector to sort
 42 */
 43 void selection_sort(vector<int>& a)
 44 {
 45    int next; // The next position to be set to the minimum
 46 
 47    for (next = 0; next < a.size() - 1; next++)
 48    {
 49       // Find the position of the minimum
 50       int min_pos = min_position(a, next, a.size() - 1);
 51       if (min_pos != next)
 52          swap(a[min_pos], a[next]);
 53    }
 54 }

 67 /**
 68    Sets the seed of the random number generator.
 69 */
 70 void rand_seed()
 71 {
 72    int seed = static_cast<int>(time(0));
 73    srand(seed);
 74 }
 75 
 76 /**
 77    Computes a random integer in a range.
 78    @param a the bottom of the range
 79    @param b the top of the range
 80    @return a random integer x, a <= x and x <= b
 81 */
 82 int rand_int(int a, int b)
 83 {
 84    return a + rand() % (b - a + 1);
 85 }
