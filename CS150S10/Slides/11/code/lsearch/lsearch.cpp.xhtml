  5 /**
  6    Finds an element in a vector
  7    @param v the vector with the elements to search
  8    @param value the value to search for
  9    @return the index of the first match, or -1 if not found
 10 */
 11 int linear_search(vector<int> v, int value)
 12 {
 13    for (int i = 0; i < v.size(); i++)
 14    {
 15       if (v[i] == value)
 16          return i;
 17    }
 18    return -1;
 19 }
