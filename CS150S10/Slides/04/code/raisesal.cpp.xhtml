  7 /**
  8    Raises an employee salary.
  9    @param e employee receiving raise

 12 void raise_salary(Employee& e, double by)
 13 {
 14    double new_salary = e.get_salary() * (1 + by / 100);
 15    e.set_salary(new_salary);
 16 }
 17
