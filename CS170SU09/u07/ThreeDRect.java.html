01: /**
02:     The ThreeDRect applet
03:     CS 170 : using the Graphics draw3DRect() method
04:     @author Stephen Gilbert
05:     @version 1.1 - February 5, 2005
06: */
07: import java.applet.*;
08: import java.awt.*;
09: 
10: public class ThreeDRect extends Applet
11: {
12:     public void paint(Graphics g)
13:     {
14:         // Clear the background
15:         g.setColor(Color.lightGray);
16:         g.fillRect(0, 0, 200, 200);
17: 
18:         // Draw the 3D rectangles
19:         // Note that these are the same color as the background
20:         g.draw3DRect( 50,   50, 75, 37, true);
21:         g.draw3DRect( 51,   51, 73, 35, true);
22:         g.draw3DRect( 52,   52, 71, 33, true);
23:         g.draw3DRect( 53,   53, 69, 31, true);
24:     }
25: }
