13: import java.util.*;
14: 
15: public class InsertArray
16: {
17:     public static void main(String[] args)
18:     {
19:         // 1. Scanner for input
20:         Scanner scan = new Scanner(System.in);
21:     
22:         // 2. Here's our array and its maximum length
23:         double[] ar = new double[100];
24:         int capacity = ar.length;
25:         int size = 0;
26:         
27:         // 3. Get the input from the user
28:         System.out.println("Enter up to " + capacity +
29:                            " positive real numbers.");
30:         System.out.println("Enter 0 or a negative number to quit.");
31:         while (size < capacity)
32:         {
33:             double num = scan.nextDouble();
34:             if (num <= 0)
35:                 break;
36:             
37:             //  Got a valid number. Where should we put it?
38:             //  Loop through and find the first number that's
39:             //  larger than the number input. Store it in "i"
40:             int i = 0;
41:             for (i = 0; i < size; i++)
42:                 if (ar[i] > num)
43:                     break;
44:             
45:             //  Move the array elements to make room for 
46:             //  the new number. Note that you must move the
47:             //  numbers starting at the end of the array
48:             for (int j = size; j > i; j--)
49:                 ar[j] = ar[j-1];
50:             
51:             // Put the current element into its position
52:             ar[i] = num;
53:             
54:             // Increment the size
55:             size++;
56:         }
57:     
58:         // 8. Display results
59:         System.out.print("ar = {");
60:         if (size == 0)
61:         {
62:             System.out.println("}");
63:         }
64:         else
65:         {
66:             for (int i = 0; i < size -1; i++)
67:                 System.out.print(ar[i] + ", ");
68:             System.out.println(ar[size-1] + "}");
69:         }
70:     }
71: }
