01: /**
02:  *  NumPad.java
03:  *  Illustrates initializing and using a simple 2D array
04:  *  Simulates a numeric keypad
05:  *  @author Stephen Gilbert
06:  *  @version 2.0, November 2005
07:  */
08: import java.awt.*;
09: import java.applet.*;
10: import java.awt.event.*;
11: 
12: public class NumPad extends Applet
13: {
14:     private Label output = new Label("", Label.RIGHT);
15:     
16:     public void init()
17:     {
18:         setFont(new Font("Monospaced", Font.BOLD, 18));
19:         
20:         Panel p = new Panel();
21:         p.setLayout(new GridLayout(4,3,3,3));
22:         
23:         String[][] keyCaps = {
24:             { "7", "8", "9" },
25:             { "4", "5", "6" },
26:             { "1", "2", "3" },
27:             { "0", "C", "." }};
28:         
29:         for (int row = 0; row < 4; row++)
30:         {
31:             for (int col = 0; col < 3; col++)
32:             {
33:                 Button b = new Button(keyCaps[row][col]);
34:                 p.add(b);
35:                 b.addActionListener(new NumHandler());
36:             }    
37:         }
38:         setLayout(new BorderLayout(5,5));
39:         add("Center", p);
40:         add("North", output);
41:     }
42:     
43:     /**
44:      * Handles button clicks for the NumPad class
45:      */
46:     class NumHandler implements ActionListener
47:     {
48:         public void actionPerformed(ActionEvent ae)
49:         {
50:             Button pressed = (Button)(ae.getSource());
51:             String cmd = pressed.getActionCommand();
52:             
53:             if ( cmd.equals("C")) output.setText("");
54:             else output.setText(output.getText() + cmd);
55:         }
56:     }
57: }
