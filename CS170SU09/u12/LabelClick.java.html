import java.awt.*;
import java.awt.event.*;
import java.applet.*;

/**
 * Shows how to use MouseAdapter to listen for clicks CS 170 example program
 *
 * @author Stephen Gilbert
 * @version 2.0, May 2009
 */

public class LabelClick extends Applet
{
    /**
     * Set up the user interface.
     */
    public void init()
    {
        setFont(new Font("SansSerif", Font.BOLD, 24));

        Label lbl = new Label("Yellow");
        lbl.setForeground(Color.yellow);
        lbl.addMouseListener(new BGChanger());
        add(lbl);

        lbl = new Label("Red");
        lbl.setForeground(Color.red);
        lbl.addMouseListener(new BGChanger());
        add(lbl);

        lbl = new Label("Green");
        lbl.setForeground(Color.green);
        lbl.addMouseListener(new BGChanger());
        add(lbl);

        lbl = new Label("Blue");
        lbl.setForeground(Color.blue);
        lbl.addMouseListener(new BGChanger());
        add(lbl);
    }

    /**
     * Mouse adapter inner class.
     */
    private class BGChanger extends MouseAdapter
    {
        public void mouseClicked(MouseEvent event)
        {
            setBackground(((Label)event.getSource()).getForeground());
        }
    }
}
