09: import java.awt.*;
10: import java.applet.*;
11: import java.awt.event.*;
12: 
13: public class PolyDraw extends Applet
14:             implements MouseListener
15: {
16:     private int[] ax = new int[100];
17:     private int[] ay = new int[100];
18:     private int n = 0;
19: 
20:     /**
21:      * Add the mouse listener to the applet
22:      */
23:     public void init()
24:     {
25:         addMouseListener(this);
26:     }
27:     
28:     /**
29:      * Draw a line using all of the points
30:      */
31:     public void paint(Graphics g)
32:     {
33:         g.drawPolyline(ax, ay, n);
34:     }
35:     
36:     public void mouseClicked(MouseEvent e)
37:     {
38:         if (n >= ax.length)
39:             n = 0; // wrap around
40: 
41:         ax[n] = e.getX();
42:         ay[n] = e.getY();
43:         n++;
44: 
45:         repaint();
46:     }
47:     
48:     // Ignore all of these 
49:     public void mousePressed(MouseEvent e) { }
50:     public void mouseReleased(MouseEvent e){ }
51:     public void mouseEntered(MouseEvent e) { }
52:     public void mouseExited(MouseEvent e)  { }
53: }
