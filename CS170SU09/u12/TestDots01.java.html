13: import java.applet.Applet;
14: import java.awt.*;
15: import java.util.*;
16: 
17: public class TestDots extends Applet
18: {
19:     /**
20:      * Create some dots and scatter them around
21:      * the applet. Use component methods to set the
22:      * size, location, and background color
23:      */
24:     public void init()
25:     {
26:         setBackground(Color.yellow);
27:         Random rand = new Random();
28:         setLayout(null);
29:         
30:         int width = getWidth();
31:         int height = getHeight();
32:         
33:         for (int i = 0; i < 10; i++)
34:         {
35:             Dot aDot = new Dot();
36:             add(aDot);
37:             aDot.setLocation(rand.nextInt(width), 
38:                              rand.nextInt(height));
39:             aDot.setForeground(new Color(rand.nextInt()));
40:             aDot.setRadius(rand.nextInt(25) + 3);
41:         }
42:     }
43: }
