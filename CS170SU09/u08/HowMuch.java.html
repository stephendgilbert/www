11: // 1. Add your import statments here
12: import  java.awt.*;
13: import  java.applet.*;
14: import  java.awt.event.*;
15: import  java.text.*;
16: 
17: public class HowMuch extends Applet
18: {
19:     // 2. Create your object attributes here
20:     //    You'll have exactly 6 objects: three Labels,
21:     //    two TextFields, and one Button
22:     private TextField  interestTF   =  new  TextField(20);
23:     private TextField  amountTF     =  new  TextField(20);
24:     private TextField  yearsToPayTF =  new  TextField(20);
25: 
26:     private Label  paymentLbl   =  new  Label("",  Label.RIGHT);
27:     private Label  totalCostLbl =  new  Label("",  Label.RIGHT);
28: 
29:     private Button calcCostBtn  =  new  Button("Calculate  Cost");
30: 
31:     public void init()
32:     {
33:         // This line simply arranges the components
34:         // Do NOT remove this line or put any code before it
35:         setLayout(new GridLayout(0, 2, 5, 5));
36:         setBackground(Color.white);
37: 
38:         // 3. Add the components to your applet. For each
39:         //    line in the finished calculator, you'll need
40:         //    to create and add a local descriptive label,
41:         //    and then add one of the fields created in
42:         //    section 2. You won't have a label for the
43:         //    last row that contains the button.
44:         add(  new  Label("Amount  of  Purchase:",  Label.RIGHT));
45:         add(amountTF);
46: 
47:         add(  new  Label("Interest  Rate  [like  7.5]  :",Label.RIGHT));
48:         add(interestTF);
49: 
50:         add(  new  Label("Years  to  Pay  :",  Label.RIGHT));
51:         add(yearsToPayTF);
52: 
53:         add(  new  Label("Monthly  Payment  :",  Label.RIGHT));
54:         add(paymentLbl);
55: 
56:         add(  new  Label("Total  Purchase  Cost  :",  Label.RIGHT));
57:         add(totalCostLbl);
58: 
59:         add(calcCostBtn);
60: 
61:         // 4. Hook-up the Java event system
62:         calcCostBtn.addActionListener(new ButtonHandler());
63:     }
64: 
65:     /**
66:      *  Responds to ActionEvents from the button
67:      */
68:     private class ButtonHandler implements ActionListener
69:     {
70:         public void actionPerformed(ActionEvent ae)
71:         {
72:             // 5. Extract text from each text field
73:             String  interestStr    =  interestTF.getText();
74:             String  amountStr      =  amountTF.getText();
75:             String  yearsToPayStr  =  yearsToPayTF.getText();
76: 
77:             // 6. Create numeric variables and convert text
78:             //  Note: these are the Java 1.1 conversion methods. You could also use
79:             //           Double.parseDouble() if you only wanted your applet to run under
80:             //           Java 2 browsers
81:             double  interest   =  Double.valueOf(interestStr).doubleValue();
82:             double  amount     =  Double.valueOf(amountStr).doubleValue();
83:             double  yearsToPay =  Double.valueOf(yearsToPayStr).doubleValue();
84: 
85:             // 7. Make the necessary calculations
86:             double  monthlyRate    =  interest  /  100  /  12;
87:             double  payment  =  (amount  *  monthlyRate)  /
88:                         (1-(Math.pow(1/(1  +  monthlyRate),  yearsToPay  *  12)));
89:             double  totalCost  =  yearsToPay  *  12  *  payment;
90: 
91:             // 8. Display the formatted results
92:             NumberFormat  fmt  =  NumberFormat.getCurrencyInstance();
93:             paymentLbl.setText(fmt.format(payment));
94:             totalCostLbl.setText(fmt.format(totalCost));
95:         }
96:     }
97: }
