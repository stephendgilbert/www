import acm.program.*;
import acm.graphics.*;
import java.awt.*;
import java.awt.event.*;
/**
 * The TwoCards program uses the GOval, GImage,
 * and GLabel classes to display two cards on
 * the screen. Uses the GPL card images available
 * from <a href='http://www.waste.org/%7Eoxymoron/cards/'>
 * http://www.waste.org/~oxymoron/cards/</a>
 * 
 * @author Stephen Gilbert
 * @version 1.0, September 2005
 */
public class TwoCards extends GraphicsProgram
{
    public void run()
    {
        // 1. Create a green "table" for our cards
        final int T_WIDTH = 200;
        final int T_HEIGHT = 150;
        GRect table = new GRect(0, 0, T_WIDTH, T_HEIGHT);
        
        // 1a. Center the table
        int appWidth = getWidth();
        int appHeight = getHeight();
        table.setLocation(appWidth/2 - T_WIDTH/2,
                          appHeight/2 - T_HEIGHT/2);

        // 1b. Fill it with a dark green color
        table.setFillColor(new Color(0, 128, 0)); // dark green
        table.setFilled(true);
        add(table);
        
        // 1c. Let's label it as well
        GLabel caption = new GLabel("Place your bet!");
        double capWidth = caption.getWidth();
        double capHeight = caption.getHeight();
        caption.setLocation(appWidth/2 - capWidth/2,
               appHeight/2 + T_HEIGHT/2 + capHeight);
        add(caption);
        
        // 2. Let's create a card using the GImage class
        GImage card1 = new GImage("cards/as.gif");
        card1.setLocation(table.getLocation());
        card1.move(15, 15);
        add(card1);
        
        // 2b. We'll deal the second card with the back showing
        GImage card2 = new GImage("cards/b.gif");
        card2.setLocation(card1.getLocation());
        card2.move(25, 10);
        add(card2);
        
        // 3. Hook up mouse listeners
        addMouseListeners();
    }
    
    // Turn over the card when clicked
    public void mouseClicked(MouseEvent e)
    {
        int x = e.getX();
        int y = e.getY();
        
        GImage card = (GImage) getElementAt(x, y);
        card.setImage("cards/7c.gif");
        card.move(card.getWidth(), -10);
    }
}
