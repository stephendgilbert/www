01: /**
02:  *  ReallyDoIt.java
03:  *
04:  *  Adds a button to an applet, and hooks it up
05:  *
06:  *  @author Stephen Gilbert
07:  *  @version 2.0, Fall 2005
08:  */
09: 
10: import java.awt.*;
11: import java.awt.event.*;
12: import java.applet.*;
13: 
14: public class ReallyDoIt extends Applet
15: {
16:     private Button doItBtn = new Button();
17:     private int timesPressed = 0;
18: 
19: 
20:     public void init()
21:     {
22:         doItBtn.setLabel(
23:             "I'm a lonely button that's never been pressed. "+
24:             "Click me you fool!");
25:         this.add(doItBtn);
26:         doItBtn.addActionListener(new ButtonListener());
27:     }
28: 
29:     /**
30:      *  ButtonListener
31:      *  Handles button events for ReallyDoIt
32:     */
33:     private class ButtonListener implements ActionListener
34:     {
35:         public void actionPerformed(ActionEvent e)
36:         {
37:             timesPressed++;
38:             String btnText = "I've been pressed " + timesPressed +
39:                              " times";
40:             doItBtn.setLabel(btnText);
41:         }
42:     }
43: 
44: }
