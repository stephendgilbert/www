01: /**
02:  * Displays a few JButtons using different constructors
03:  * 
04:  * @author Stephen Gilbert
05:  * @version 1.0, September 2005
06:  */
07: import java.awt.*;
08: import java.awt.event.*;
09: import javax.swing.*;
10: 
11: public class JButton01 extends JPanel
12: {
13:     private JLabel outputLabel = new JLabel("Output will appear here!!!");
14:     
15:     /**
16:      * Constructor for objects of class JButton01
17:      */
18:     public JButton01()
19:     {
20:         setBackground(new Color(196, 196, 255));
21:         
22:        // 1. Create three JButtons
23:        JButton btn1 = new JButton(new ImageIcon("Car1.GIF"));
24:        JButton btn2 = new JButton("Plain Text");
25:        JButton btn3 = new JButton("Car2.GIF",
26:                       new ImageIcon("Car2.GIF"));
27: 
28:        // 2. Set mnemonics & tooltips for JButtons
29:        btn1.setMnemonic(KeyEvent.VK_F1);
30:        btn1.setToolTipText("Click here to order a blue car");
31:        btn2.setMnemonic('P');
32:        btn2.setToolTipText("This is only text");
33:        btn3.setMnemonic(KeyEvent.VK_1); // Alt-1
34:        btn3.setToolTipText("Another helpful message");
35: 
36:        // 3. Set the action commands for the buttons
37:        btn1.setActionCommand("cut");
38:        btn2.setActionCommand("copy");
39:        btn3.setActionCommand("paste");
40: 
41:        // 4. Add the action listener to the buttons
42:        ButtonListener listener = new ButtonListener();
43:        btn1.addActionListener(listener);
44:        btn2.addActionListener(listener);
45:        btn3.addActionListener(listener);
46: 
47:        // 5. Add the buttons to the application
48:        add(btn1);
49:        add(btn2);
50:        add(btn3);
51:        add(outputLabel);
52:        
53:     }
54:     
55:     /**
56:      * The class responds to clicks from a JButton
57:      */
58:     private class ButtonListener implements ActionListener
59:     {
60:         public void actionPerformed(ActionEvent ae)
61:         {
62:             outputLabel.setText("The button clicked says " + 
63:                                  ae.getActionCommand());
64:         }
65:     }
66:     
67:     /**
68:      * The main() method creates the JFrame and the application panel
69:      */
70:     public static void main(String[] args)
71:     {
72:         JFrame frame = new JFrame("JButton01");
73:         frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
74:         frame.getContentPane().add(new JButton01());
75:         frame.setSize(350, 150);
76:         frame.setVisible(true);
77:     }
78: }
